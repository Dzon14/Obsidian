$$\begin{align} y(t) = asin(\omega t + \theta) = \lvert G(i \omega)\rvert sin(\omega t+ \theta) \\ \\ \theta = arg \ G(iw) \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \end{align} $$
